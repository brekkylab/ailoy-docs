278    The `Agent` class provides a high-level interface for interacting with large language models (LLMs) in Ailoy.
279    It abstracts the underlying runtime and VM logic, allowing users to easily send queries and receive streaming
280    responses.

332    def define(self, model_name: ModelName, api_key: Optional[str] = None, attrs: dict[str, Any] = dict()) -> None:
333        """
334        Initializes the agent by defining its model in the runtime.
335        This must be called before running the agent. If already initialized, this is a no-op.
336        :param model_name: The name of the LLM model to use.
337        :param api_key: (web agent only) The API key for AI API.
338        :param attrs: Additional initialization parameters (for `define_component` runtime call)
339        """
340        if self._component_state.valid:
341            return
342
343        if model_name not in model_descriptions:
344            raise ValueError(f"Model `{model_name}` not supported")
345
346        model_desc = model_descriptions[model_name]
347
348        # Add model name into attrs
349        if "model" not in attrs:
350            attrs["model"] = model_desc.model_id
351
352        # Add API key
353        if api_key:
354            attrs["api_key"] = api_key
355
356        # Call runtime's define
357        self._runtime.define(
358            model_descriptions[model_name].component_type,
359            self._component_state.name,
360            attrs,
361        )
362
363        # Mark as defined
364        self._component_state.valid = True

366    def delete(self) -> None:
367        """
368        Deinitializes the agent and releases resources in the runtime.
369        This should be called when the agent is no longer needed. If already deinitialized, this is a no-op.
370        """
371        if not self._component_state.valid:
372            return
373        self._runtime.delete(self._component_state.name)
374        if len(self._messages) > 0 and self._messages[0].role == "system":
375            self._messages = [self._messages[0]]
376        else:
377            self._messages = []
378        self._component_state.valid = False

380    def run(
381        self,
382        message: str,
383        enable_reasoning: Optional[bool] = None,
384        ignore_reasoning_messages: Optional[bool] = None,
385    ) -> Generator[AgentResponse, None, None]:
386        """
387        Runs the agent with a new user message and yields streamed responses.
388
389        :param message: The user message to send to the model.
390        :param enable_reasoning: If True, enables reasoning capabilities (default: True).
391        :param ignore_reasoning_messages: If True, reasoning steps are not included in the response stream.
392        :yield: AgentResponse output of the LLM inference or tool calls
393        """
394        self._messages.append(UserMessage(role="user", content=message))
395
396        while True:
397            infer_args = {
398                "messages": [msg.model_dump() for msg in self._messages],
399                "tools": [{"type": "function", "function": t.desc.model_dump()} for t in self._tools],
400            }
401            if enable_reasoning is not None:
402                infer_args["enable_reasoning"] = enable_reasoning
403            if ignore_reasoning_messages is not None:
404                infer_args["ignore_reasoning_messages"] = ignore_reasoning_messages
405
406            for resp in self._runtime.call_iter_method(self._component_state.name, "infer", infer_args):
407                delta = MessageDelta.model_validate(resp)
408
409                if delta.finish_reason is None:
410                    output_msg = AIOutputTextMessage.model_validate(delta.message)
411                    yield AgentResponseOutputText(
412                        type="reasoning" if output_msg.reasoning else "output_text",
413                        end_of_turn=False,
414                        role="assistant",
415                        content=output_msg.content,
416                    )
417                    continue
418
419                if delta.finish_reason == "tool_calls":
420                    tool_call_message = AIToolCallMessage.model_validate(delta.message)
421                    self._messages.append(tool_call_message)
422
423                    for tool_call in tool_call_message.tool_calls:
424                        yield AgentResponseToolCall(
425                            type="tool_call",
426                            end_of_turn=True,
427                            role="assistant",
428                            content=tool_call,
429                        )
430
431                    tool_call_results: list[ToolCallResultMessage] = []
432
433                    def run_tool(tool_call: ToolCall):
434                        tool_ = next(
435                            (t for t in self._tools if t.desc.name == tool_call.function.name),
436                            None,
437                        )
438                        if not tool_:
439                            raise RuntimeError("Tool not found")
440                        resp = tool_.call(self._runtime, tool_call.function.arguments)
441                        return ToolCallResultMessage(
442                            role="tool",
443                            name=tool_call.function.name,
444                            tool_call_id=tool_call.id,
445                            content=json.dumps(resp),
446                        )
447
448                    tool_call_results = [run_tool(tc) for tc in tool_call_message.tool_calls]
449
450                    for result_msg in tool_call_results:
451                        self._messages.append(result_msg)
452                        yield AgentResponseToolCallResult(
453                            type="tool_call_result",
454                            end_of_turn=True,
455                            role="tool",
456                            content=result_msg,
457                        )
458
459                    # Run infer again with new messages
460                    break
461
462                if delta.finish_reason in ["stop", "length", "error"]:
463                    output_msg = AIOutputTextMessage.model_validate(delta.message)
464                    yield AgentResponseOutputText(
465                        type="reasoning" if output_msg.reasoning else "output_text",
466                        end_of_turn=False,
467                        role="assistant",
468                        content=output_msg.content,
469                    )
470
471                    # finish this Generator
472                    return

494    def add_universal_tool(self, tool_def: UniversalToolDefinition) -> bool:
495        """
496        Adds a universal tool.
497
498        :param tool_def: The universal tool definition.
499        :returns: True if the tool was successfully added.
500        :raises ValueError: If the tool type is not "universal" or required inputs are missing.
501        """
502        if tool_def.type != "universal":
503            raise ValueError('Tool type is not "universal"')
504
505        def call(runtime: Runtime, inputs: dict[str, Any]) -> Any:
506            required = tool_def.description.parameters.required or []
507            for param_name in required:
508                if param_name not in inputs:
509                    raise ValueError(f'Parameter "{param_name}" is required but not provided')
510
511            output = runtime.call(tool_def.description.name, inputs)
512            if tool_def.behavior.output_path is not None:
513                output = jmespath.search(tool_def.behavior.output_path, output)
514
515            return output
516
517        return self.add_tool(Tool(desc=tool_def.description, call_fn=call))

519    def add_restapi_tool(
520        self,
521        tool_def: RESTAPIToolDefinition,
522        authenticator: Optional[Union[ToolAuthenticator, Callable[[dict[str, Any]], dict[str, Any]]]] = None,
523    ) -> bool:
524        """
525        Adds a REST API tool that performs external HTTP requests.
526
527        :param tool_def: REST API tool definition.
528        :param authenticator: Optional authenticator to inject into the request.
529        :returns: True if the tool was successfully added.
530        :raises ValueError: If the tool type is not "restapi".
531        """
532        if tool_def.type != "restapi":
533            raise ValueError('Tool type is not "restapi"')
534
535        behavior = tool_def.behavior
536
537        def call(_: Runtime, inputs: dict[str, Any]) -> Any:
538            def render_template(template: str, context: dict[str, Any]) -> tuple[str, list[str]]:
539                import re
540
541                variables = set()
542
543                def replacer(match: re.Match):
544                    key = match.group(1).strip()
545                    variables.add(key)
546                    return str(context.get(key, f"{{{key}}}"))
547
548                rendered_url = re.sub(r"\$\{\s*([^}\s]+)\s*\}", replacer, template)
549                return rendered_url, list(variables)
550
551            # Handle path parameters
552            url, path_vars = render_template(behavior.base_url, inputs)
553
554            # Handle body
555            if behavior.body is not None:
556                body, body_vars = render_template(behavior.body, inputs)
557            else:
558                body, body_vars = None, []
559
560            # Handle query parameters
561            query_params = {k: v for k, v in inputs.items() if k not in set(path_vars + body_vars)}
562
563            # Construct a full URL
564            full_url = urlunparse(urlparse(url)._replace(query=urlencode(query_params)))
565
566            # Construct a request payload
567            request = {
568                "url": full_url,
569                "method": behavior.method,
570                "headers": behavior.headers,
571            }
572            if body:
573                request["body"] = body
574
575            # Apply authentication
576            if authenticator is not None:
577                request = authenticator.apply(request)
578
579            # Call HTTP request
580            output = None
581            resp = self._runtime.call("http_request", request)
582            output = json.loads(resp["body"])
583
584            # Parse output path if defined
585            if behavior.output_path is not None:
586                output = jmespath.search(tool_def.behavior.output_path, output)
587
588            return output
589
590        return self.add_tool(Tool(desc=tool_def.description, call_fn=call))

592    def add_tools_from_preset(self, preset_name: str, authenticator: Optional[ToolAuthenticator] = None):
593        """
594        Loads tools from a predefined JSON preset file.
595
596        :param preset_name: Name of the tool preset.
597        :param authenticator: Optional authenticator to use for REST API tools.
598        :raises ValueError: If the preset file is not found.
599        """
600        tool_presets_path = Path(__file__).parent / "presets" / "tools"
601        preset_json = tool_presets_path / f"{preset_name}.json"
602        if not preset_json.exists():
603            raise ValueError(f'Tool preset "{preset_name}" does not exist')
604
605        data: dict[str, dict[str, Any]] = json.loads(preset_json.read_text())
606        for tool_name, tool_def in data.items():
607            tool_type = tool_def.get("type", None)
608            if tool_type == "universal":
609                self.add_universal_tool(UniversalToolDefinition.model_validate(tool_def))
610            elif tool_type == "restapi":
611                self.add_restapi_tool(RESTAPIToolDefinition.model_validate(tool_def), authenticator=authenticator)
612            else:
613                warnings.warn(f'Tool type "{tool_type}" is not supported. Skip adding tool "{tool_name}".')

615    def add_mcp_tool(self, params: mcp.StdioServerParameters, tool: mcp_types.Tool):
616        """
617        Adds a tool from an MCP (Model Context Protocol) server.
618
619        :param params: Parameters for connecting to the MCP stdio server.
620        :param tool: Tool metadata as defined by MCP.
621        :returns: True if the tool was successfully added.
622        """
623        from mcp.client.stdio import stdio_client
624
625        def call(_: Runtime, inputs: dict[str, Any]) -> Any:
626            async def _inner():
627                async with stdio_client(params, errlog=subprocess.STDOUT) as streams:
628                    async with mcp.ClientSession(*streams) as session:
629                        await session.initialize()
630
631                        result = await session.call_tool(tool.name, inputs)
632                        contents: list[str] = []
633                        for item in result.content:
634                            if isinstance(item, mcp_types.TextContent):
635                                contents.append(item.text)
636                            elif isinstance(item, mcp_types.ImageContent):
637                                contents.append(item.data)
638                            elif isinstance(item, mcp_types.EmbeddedResource):
639                                if isinstance(item.resource, mcp_types.TextResourceContents):
640                                    contents.append(item.resource.text)
641                                else:
642                                    contents.append(item.resource.blob)
643
644                        return contents
645
646            return run_async(_inner())
647
648        desc = ToolDescription(name=tool.name, description=tool.description, parameters=tool.inputSchema)
649        return self.add_tool(Tool(desc=desc, call_fn=call))

651    def add_tools_from_mcp_server(self, params: mcp.StdioServerParameters, tools_to_add: Optional[list[str]] = None):
652        """
653        Fetches tools from an MCP stdio server and registers them with the agent.
654
655        :param params: Parameters for connecting to the MCP stdio server.
656        :param tools_to_add: Optional list of tool names to add. If None, all tools are added.
657        :returns: list of all tools returned by the server.
658        """
659        from mcp.client.stdio import stdio_client
660
661        async def _inner():
662            async with stdio_client(params, errlog=subprocess.STDOUT) as streams:
663                async with mcp.ClientSession(*streams) as session:
664                    await session.initialize()
665                    resp = await session.list_tools()
666                    for tool in resp.tools:
667                        if tools_to_add is None or tool.name in tools_to_add:
668                            self.add_mcp_tool(params, tool)
669                    return resp.tools
670
671        tools = run_async(_inner())
672        return tools

414                        end_of_turn=False,
415                        role="assistant",
416                        content=output_msg.content,
417                    )
418                    continue
419
420                if delta.finish_reason == "tool_calls":
421                    tool_call_message = AIToolCallMessage.model_validate(delta.message)
422                    self._messages.append(tool_call_message)
423
424                    for tool_call in tool_call_message.tool_calls:
425                        yield AgentResponseToolCall(
426                            type="tool_call",

521        self,
522        tool_def: RESTAPIToolDefinition,
523        authenticator: Optional[Union[ToolAuthenticator, Callable[[dict[str, Any]], dict[str, Any]]]] = None,
524    ) -> bool:
525        """
526        Adds a REST API tool that performs external HTTP requests.
527
528        :param tool_def: REST API tool definition.
529        :param authenticator: Optional authenticator to inject into the request.
530        :returns: True if the tool was successfully added.
531        :raises ValueError: If the tool type is not "restapi".

534            raise ValueError('Tool type is not "restapi"')

536        behavior = tool_def.behavior
537
538        def call(_: Runtime, inputs: dict[str, Any]) -> Any:
539            def render_template(template: str, context: dict[str, Any]) -> tuple[str, list[str]]:
540                import re
541
542                variables = set()
543
544                def replacer(match: re.Match):
545                    key = match.group(1).strip()
546                    variables.add(key)
547                    return str(context.get(key, f"{{{key}}}"))
548
549                rendered_url = re.sub(r"\$\{\s*([^}\s]+)\s*\}", replacer, template)
550                return rendered_url, list(variables)
551
552            # Handle path parameters
553            url, path_vars = render_template(behavior.base_url, inputs)
554
555            # Handle body
556            if behavior.body is not None:
557                body, body_vars = render_template(behavior.body, inputs)
558            else:
559                body, body_vars = None, []
560
561            # Handle query parameters
562            query_params = {k: v for k, v in inputs.items() if k not in set(path_vars + body_vars)}
563
564            # Construct a full URL
565            full_url = urlunparse(urlparse(url)._replace(query=urlencode(query_params)))
566
567            # Construct a request payload
568            request = {
569                "url": full_url,
570                "method": behavior.method,
571                "headers": behavior.headers,
572            }
573            if body:
574                request["body"] = body
575
576            # Apply authentication
577            if authenticator is not None:
578                request = authenticator.apply(request)
579
580            # Call HTTP request
581            output = None
582            resp = self._runtime.call("http_request", request)
583            output = json.loads(resp["body"])
584
585            # Parse output path if defined
586            if behavior.output_path is not None:
587                output = jmespath.search(tool_def.behavior.output_path, output)
588
589            return output
590
591        return self.add_tool(Tool(desc=tool_def.description, call_fn=call))
